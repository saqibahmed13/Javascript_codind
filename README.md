Converting array to objects 

function arrayToObject(arr){
    let obj = {};
    for(let i = 0; i<arr.length;i++){
        obj[i]=arr[i];
    }
    return obj;
}
let arr = [1,2,3,4,5];
console.log(arr);
let obj = arrayToObject(arr);
console.log(obj);
